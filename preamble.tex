\def\@part[#1]#2
{%
    \refstepcounter{part}%
    {%
        \parindent \z@ \raggedright \interlinepenalty \@M
        \normalfont \Large\bfseries\raggedright
        \partname\nobreakspace\thepart : \nobreakspace #2 %\markboth{}{}\par
    }%
    \nobreak \vskip 1.3ex \@afterheading%
}

\def\@maketitle
{%

  \let \footnote \thanks
    {\Large\bf \@title \par}%
    \vskip 0.5em%
    {\large
      \lineskip 1em%
      \begin{tabular}[t]{c}%
        \@author
      \end{tabular}\par}%
    \vskip 0.5em%
    {\large \@date}%
  \end{center}%
  \par
  %\vskip 1.5em
}
